function obStep(n){
  document.querySelectorAll('#s-onboard .onboard').forEach(o=>o.style.display='none');

function obComplete(){
  document.querySelectorAll('.sc').forEach(s=>s.classList.remove('on'));
  document.getElementById('s-home').classList.add('on');
  document.getElementById('bnav').style.display='flex';
  showToast('Welcome to MediConnect, Rahul','ok');
}
function go(id){
  document.querySelectorAll('.sc').forEach(s=>s.classList.remove('on'));
  document.getElementById(id).classList.add('on');
  const map={'s-home':'s-home','s-book':'s-book','s-chat':'s-chat','s-meds':'s-meds','s-records':'s-records','s-pay':'s-meds'};
  const bp=map[id]||id;
  document.querySelectorAll('.bi').forEach(b=>b.classList.toggle('on',b.dataset.s===bp));
  if(id==='s-book'){document.getElementById('book-s1').style.display='flex';document.getElementById('book-s2').style.display='none';}
  if(id==='s-pay'){document.getElementById('pay-s1').style.display='flex';document.getElementById('pay-s2').style.display='none';}
}

  document.querySelectorAll('.slot-grid .slot').forEach(s=>{if(!s.classList.contains('gone'))s.classList.remove('on');});

}

}
function switchTab(el){

function qtyChg(btn,delta){

  if(n<1)n=1;

}
